00092 #define OBJ_TRIMESH 0x4100              /*OK Triangular mesh, contains chunks for 3d mesh info - Lenght: 0 + sub chunks */

00094                                                                 Lenght:
00095                                                                         1 x unsigned short (number of vertices)

00100                                                                 Lenght:
00101                                                                         1 x unsigned short (number of vertices)

00103                                                                         + sub chunks */
00104 #define TRI_MATERIAL 0x4130             /* #FALTA# MSH_MAT_GROUP mesh_material_group - cstr material_name; short nfaces; short facenum[nfaces]; */

00110 #define TRI_SMOOTH 0x4150               /* #POSIBLE# SMOOTH_GROUP - short grouplist[n]; (determined by length, seems to be 4 per face) */
